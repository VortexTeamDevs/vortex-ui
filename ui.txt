-- Venus UI ile Auto Buy Sistemi
-- Modern & Çalışan Library

local Venus = loadstring(game:HttpGet('https://raw.githubusercontent.com/Stefanuk12/Venus/main/Loader.lua'))()

-- Pencere oluştur
local Window = Venus:Init('Auto Buy System', 'v1.0')

-- Sistem değişkenleri
local AutoBuy = {
    Enabled = false,
    Tool = "Blackstone Platform",
    Speed = 50,
    Total = 0
}

-- Tab'ları oluştur
local MainTab = Window:Tab('Ana Sayfa')
local AutoTab = Window:Tab('Auto Buy')
local StatsTab = Window:Tab('İstatistikler')

-- Ana Sayfa
local MainSection = MainTab:Section('Hızlı Kontroller')
MainSection:Label('Auto Buy System v1.0')
MainSection:Label('Premium Auto Buy Sistemi')

MainSection:Button('10x Hızlı Al', function()
    for i = 1, 10 do
        local args = {"Blackstone Platform", 0}
        game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
        AutoBuy.Total = AutoBuy.Total + 1
        task.wait(0.05)
    end
    Venus:Notification('10x Satın Alındı!', '10 adet platform alındı.', 3)
end)

MainSection:Button('50x Hızlı Al', function()
    for i = 1, 50 do
        local args = {"Blackstone Platform", 0}
        game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
        AutoBuy.Total = AutoBuy.Total + 1
        task.wait(0.03)
    end
    Venus:Notification('50x Satın Alındı!', '50 adet platform alındı.', 3)
end)

-- Auto Buy Tab
local AutoSection = AutoTab:Section('Otomatik Sistem')
local Toggle = AutoSection:Toggle('Auto Buy Açık', false, function(state)
    AutoBuy.Enabled = state
    if state then
        Venus:Notification('Auto Buy Başlatıldı', 'Sistem çalışıyor...', 2)
        
        -- Auto buy döngüsü
        while AutoBuy.Enabled do
            local args = {"Blackstone Platform", 0}
            local success = pcall(function()
                game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
                AutoBuy.Total = AutoBuy.Total + 1
            end)
            
            if not success then
                Venus:Notification('Hata!', 'Satın alma başarısız.', 2)
            end
            
            task.wait((100 - AutoBuy.Speed) / 100)
        end
    else
        Venus:Notification('Auto Buy Durduruldu', 'Sistem durdu.', 2)
    end
end)

AutoSection:Slider('Hız Ayarı', 1, 100, 50, function(value)
    AutoBuy.Speed = value
end)

AutoSection:Button('Acil Durdur', function()
    AutoBuy.Enabled = false
    Toggle:Set(false)
    Venus:Notification('ACİL DURDUR!', 'Tüm sistemler durduruldu.', 3)
end)

-- İstatistikler Tab
local StatsSection = StatsTab:Section('Sistem İstatistikleri')
StatsSection:Label('Toplam Satın Alınan: 0')
StatsSection:Label('Mevcut Hız: 50')
StatsSection:Label('Sistem Durumu: Kapalı')

StatsSection:Button('İstatistikleri Yenile', function()
    Venus:Notification('Yenilendi', 'İstatistikler güncellendi.', 1)
end)

StatsSection:Button('Sayaç Sıfırla', function()
    AutoBuy.Total = 0
    Venus:Notification('Sıfırlandı', 'Tüm veriler sıfırlandı.', 2)
end)

-- İstatistik güncelleme
coroutine.wrap(function()
    while true do
        -- İstatistikleri güncelle
        for i, v in pairs(StatsSection:GetChildren()) do
            if v.Text then
                if v.Text:find("Toplam") then
                    v:Set('Toplam Satın Alınan: ' .. AutoBuy.Total)
                elseif v.Text:find("Mevcut Hız") then
                    v:Set('Mevcut Hız: ' .. AutoBuy.Speed)
                elseif v.Text:find("Sistem Durumu") then
                    v:Set('Sistem Durumu: ' .. (AutoBuy.Enabled and 'Açık' or 'Kapalı'))
                end
            end
        end
        task.wait(1)
    end
end)()

-- Keybind sistemi
game:GetService("UserInputService").InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.F then
        AutoBuy.Enabled = not AutoBuy.Enabled
        Toggle:Set(AutoBuy.Enabled)
        
        Venus:Notification(
            AutoBuy.Enabled and 'Auto Buy Açıldı' or 'Auto Buy Kapandı',
            'F tuşu ile kontrol edildi.',
            1
        )
    end
end)

-- Başlangıç bildirimi
Venus:Notification(
    'Auto Buy Sistemi Hazır!',
    'F tuşu ile açıp kapatabilirsin.',
    4
)

print("Venus UI Auto Buy sistemi yüklendi!")
print("F tuşu: Auto Buy Aç/Kapa")
print("Özellikler: Hızlı alım, Auto buy, İstatistikler")
