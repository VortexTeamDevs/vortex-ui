-- Simple Auto Buy UI
-- Kendi Mini Library'miz

local AutoBuy = {
    Enabled = false,
    Tool = "Blackstone Platform", 
    Speed = 50,
    Total = 0
}

-- Basit UI Oluştur
local screenGui = Instance.new("ScreenGui")
screenGui.Parent = game.CoreGui

mainFrame.Position = UDim2.new(0.5, -150, 0.5, -100)

local stroke = Instance.new("UIStroke")

stroke.Thickness = 1

-- Başlık
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 30)
title.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Text = "Auto Buy System"
title.TextSize = 14
title.Font = Enum.Font.GothamBold

-- Auto Buy Butonu
local autoBtn = Instance.new("TextButton")
autoBtn.Size = UDim2.new(1, -20, 0, 40)
autoBtn.Position = UDim2.new(0, 10, 0, 40)

quick10.BackgroundColor3 = Color3.fromRGB(0, 100, 255)
quick10.TextColor3 = Color3.fromRGB(255, 255, 255)
quick10.Text = "10x AL"
quick10.TextSize = 11
quick10.Font = Enum.Font.Gotham
quick10.Parent = mainFrame

local quick50 = Instance.new("TextButton")
quick50.Size = UDim2.new(0.45, 0, 0, 30)

quick50.TextColor3 = Color3.fromRGB(255, 255, 255)

-- Bilgi Label
local infoLabel = Instance.new("TextLabel")
infoLabel.Size = UDim2.new(1, -20, 0, 40)
infoLabel.Position = UDim2.new(0, 10, 0, 130)
infoLabel.BackgroundTransparency = 1
infoLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
infoLabel.Text = "Toplam: 0\nF Tuşu: Aç/Kapa"
infoLabel.TextSize = 11
infoLabel.Font = Enum.Font.Gotham
infoLabel.TextXAlignment = Enum.TextXAlignment.Left
infoLabel.Parent = mainFrame

-- Buton Fonksiyonları
autoBtn.MouseButton1Click:Connect(function()
    AutoBuy.Enabled = not AutoBuy.Enabled
    
    if AutoBuy.Enabled then
        autoBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 83)
        autoBtn.Text = "AUTO BUY: AÇIK"
        
        -- Auto buy döngüsü
        while AutoBuy.Enabled do
            local args = {AutoBuy.Tool, 0}
            game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
            AutoBuy.Total = AutoBuy.Total + 1
            infoLabel.Text = "Toplam: " .. AutoBuy.Total .. "\nF Tuşu: Aç/Kapa"
            task.wait(0.1)
        end
    else
        autoBtn.BackgroundColor3 = Color3.fromRGB(255, 60, 60)
        autoBtn.Text = "AUTO BUY: KAPALI"
    end
end)

quick10.MouseButton1Click:Connect(function()
    for i = 1, 10 do
        local args = {AutoBuy.Tool, 0}
        game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
        AutoBuy.Total = AutoBuy.Total + 1
        task.wait(0.05)
    end
    infoLabel.Text = "Toplam: " .. AutoBuy.Total .. "\n10x Alındı!"
end)

quick50.MouseButton1Click:Connect(function()
    for i = 1, 50 do
        local args = {AutoBuy.Tool, 0}
        game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
        AutoBuy.Total = AutoBuy.Total + 1
        task.wait(0.03)
    end
    infoLabel.Text = "Toplam: " .. AutoBuy.Total .. "\n50x Alındı!"
end)

-- Keybind Sistemi
game:GetService("UserInputService").InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.F then
        AutoBuy.Enabled = not AutoBuy.Enabled
        
        if AutoBuy.Enabled then
            autoBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 83)
            autoBtn.Text = "AUTO BUY: AÇIK"
            
            while AutoBuy.Enabled do
                local args = {AutoBuy.Tool, 0}
                game:GetService("ReplicatedStorage"):WaitForChild("PurchaseTool"):FireServer(unpack(args))
                AutoBuy.Total = AutoBuy.Total + 1
                infoLabel.Text = "Toplam: " .. AutoBuy.Total .. "\nF Tuşu: Aç/Kapa"
                task.wait(0.1)
            end
        else
            autoBtn.BackgroundColor3 = Color3.fromRGB(255, 60, 60)
            autoBtn.Text = "AUTO BUY: KAPALI"
        end
    end
end)

-- Sürükleme Sistemi
local dragging = false
local dragInput, dragStart, startPos

mainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)

mainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end
end)

game:GetService("UserInputService").InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
